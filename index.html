    //ResourceSpawner2.cs
    using UnityEngine;
    using System.Collections.Generic;

    public class ResourceSpawner2 : MonoBehaviour
    {
        [Header("Resource Configuration")]
        public GameObject stonePrefab;
        public GameObject manaPrefab;
        public GameObject foodPrefab;
        public int numberOfEachResource = 30;

        [Header("Tree Configuration")]
        public GameObject treePrefab;
        public int totalTrees = 5000;
        public int minTreesPerCluster = 20;
        public int maxTreesPerCluster = 100;

        [Header("Terrain Configuration")]
        public Terrain terrain;
        public float padding = 10f;

        [Header("Vertical Bounds for Resources and Trees")]
        public float resourceMinY = 25f;
        public float resourceMaxY = 35f;
        public float treeMinY = 25f;
        public float treeMaxY = 35f;

        [Header("Distance Constraints")]
        public float noTreeNearResource = 10f;
        public float noTreeNearMainBase = 50f;
        public float noSpawnNearDepot = 50f;
        public float minResourceDistance = 20f;

        private GameObject treeParent;
        private GameObject stoneParent;
        private GameObject manaParent;
        private GameObject foodParent;

        private List<Vector3> resourcePositions = new List<Vector3>();
        private Vector3 mainBasePos;
        private Vector3[] depotPositions;
    

        void Awake()
        {
            GameObject mainBase = GameObject.FindGameObjectWithTag("MainBase");
            if (mainBase != null)
            {
                mainBasePos = mainBase.transform.position;

                var depots = GameObject.FindGameObjectsWithTag("StorageDepot");
                depotPositions = new Vector3[depots.Length];
                for (int i = 0; i < depots.Length; i++)
                    depotPositions[i] = depots[i].transform.position;
            }
            else
            {
                Debug.LogError("ResourceSpawner2: MainBase not found in the scene. Please tag your main base with 'MainBase'.");
            }
        }
                                            

        void Start()
        {
            if (terrain == null)
            {
                terrain = Terrain.activeTerrain;
                if (terrain == null)
                {
                    Debug.LogError("No active terrain found. Please assign a Terrain in the Inspector.");
                    return;
                }
            }

            InitializeResourceParents();
            SpawnResources(stonePrefab, ResourceType2.Stone);
            SpawnResources(manaPrefab, ResourceType2.Mana);
            SpawnResources(foodPrefab, ResourceType2.Food);
            SpawnTrees();
        }
                                        

        void SpawnResources(GameObject prefab, ResourceType2 type)
        {
            if (prefab == null)
            {
                Debug.LogError($"ResourceSpawner2: {type} prefab is not assigned.");
                return;
            }

            List<Vector3> spawnPositions = GenerateResourcePositions(numberOfEachResource, type);

            foreach (Vector3 pos in spawnPositions)
            {
                GameObject deposit = Instantiate(prefab, pos, Quaternion.identity, GetParentForResource(type).transform);
                deposit.transform.position = pos;

                ResourceDeposit2 depositScript = deposit.GetComponent<ResourceDeposit2>();
                if (depositScript != null)
                {
                    depositScript.resourceType = type;
                }
                else
                {
                    Debug.LogError($"ResourceSpawner2: {prefab.name} does not have a ResourceDeposit2 component.");
                }

                resourcePositions.Add(pos);
            }
        }                                            

        GameObject GetParentForResource(ResourceType2 type)
        {
            switch (type)
            {
                case ResourceType2.Stone:
                    return stoneParent;
                case ResourceType2.Mana:
                    return manaParent;
                case ResourceType2.Food:
                    return foodParent;
                default:
                    Debug.LogWarning($"ResourceSpawner2: Unknown resource type '{type}'. Using root as parent.");
                    return this.gameObject;
            }
        }                                            

        List<Vector3> GenerateResourcePositions(int count, ResourceType2 type)
            {
                List<Vector3> positions = new List<Vector3>();
                int safetyLimit = count * 10;
                int attempts = 0;
        
                while (positions.Count < count && attempts < safetyLimit)
                {
                    attempts++;
        
                    Vector3 candidate = GetRandomTerrainPosition();
                    float y = terrain.SampleHeight(candidate) + terrain.transform.position.y;
        
                    if (y < resourceMinY || y > resourceMaxY)
                        continue;
        
                    candidate.y = y;
        
                    if (candidate.x < terrain.transform.position.x + padding ||
                        candidate.x > terrain.transform.position.x + terrain.terrainData.size.x - padding ||
                        candidate.z < terrain.transform.position.z + padding ||
                        candidate.z > terrain.transform.position.z + terrain.terrainData.size.z - padding)
                        continue;
        
                    if (Vector3.Distance(candidate, mainBasePos) < noTreeNearMainBase)
                        continue;
        
                    bool tooCloseDepot = false;
                    foreach (var dp in depotPositions)
                        if (Vector3.Distance(candidate, dp) < noSpawnNearDepot)
                        {
                            tooCloseDepot = true;
                            break;
                        }
                    if (tooCloseDepot)
                        continue;
        
                    bool tooClose = false;
                    foreach (Vector3 existingPos in resourcePositions)
                    {
                        if (Vector3.Distance(candidate, existingPos) < minResourceDistance)
                        {
                            tooClose = true;
                            break;
                        }
                    }
                    if (tooClose)
                        continue;
        
                    positions.Add(candidate);
                }
        
                if (positions.Count < count)
                {
                    Debug.LogWarning($"ResourceSpawner2: Only spawned {positions.Count} out of {count} requested {type} resources due to constraints.");
                }
        
                return positions;
            }                                            

        void SpawnTrees()
        {
            if (treePrefab == null)
            {
                Debug.LogError("ResourceSpawner2: treePrefab is not assigned.");
                return;
            }

            treeParent = new GameObject("TREES");
            treeParent.transform.parent = this.transform; // Optional: set as child of ResourceSpawner2

            int treesSpawned = 0;
            int safetyLimit = 10000;
            int attempts = 0;

            while (treesSpawned < totalTrees && attempts < safetyLimit)
            {
                attempts++;

                Vector3 clusterCenter;
                if (!FindValidPositionForTrees(out clusterCenter))
                {
                    if (attempts >= safetyLimit)
                    {
                        Debug.LogWarning("ResourceSpawner2: Unable to find valid positions for more trees.");
                        break;
                    }
                    continue; // Try again
                }

                int clusterCount = Random.Range(minTreesPerCluster, maxTreesPerCluster + 1);
                if (treesSpawned + clusterCount > totalTrees)
                {
                    clusterCount = totalTrees - treesSpawned;
                }

                for (int i = 0; i < clusterCount; i++)
                {
                    Vector3 treePos;
                    if (FindValidTreePositionAround(clusterCenter, out treePos))
                    {
                        GameObject tree = Instantiate(treePrefab, treePos, Quaternion.identity, treeParent.transform);
                        ResourceDeposit2 rd = tree.GetComponent<ResourceDeposit2>();
                        if (rd != null)
                        {
                            rd.resourceType = ResourceType2.Wood;
                        }
                        else
                        {
                            Debug.LogError("Tree prefab has no ResourceDeposit2 component!");
                        }
                        treesSpawned++;
                        if (treesSpawned >= totalTrees) break;
                    }
                }
            }

            Debug.Log($"ResourceSpawner2: Spawned {treesSpawned} trees in clusters.");
        }                                            

        bool FindValidTreePositionAround(Vector3 clusterCenter, out Vector3 treePos)
        {
            float clusterRadius = 30f;

            int maxAttempts = 1000;
            for (int i = 0; i < maxAttempts; i++)
            {
                Vector2 offset = Random.insideUnitCircle * clusterRadius;
                Vector3 candidate = new Vector3(clusterCenter.x + offset.x, 0, clusterCenter.z + offset.y);
                float y = terrain.SampleHeight(candidate) + terrain.transform.position.y;

                if (y < treeMinY || y > treeMaxY)
                    continue;

                candidate.y = y;

                if (Vector3.Distance(candidate, mainBasePos) < noTreeNearMainBase)
                    continue;

                bool nearDepot = false;
                foreach (var dp in depotPositions)
                    if (Vector3.Distance(candidate, dp) < noSpawnNearDepot)
                    {
                        nearDepot = true;
                        break;
                    }
                if (nearDepot)
                    continue;

                if (!IsFarFromResources(candidate, noTreeNearResource))
                    continue;

                treePos = candidate;
                return true;
            }

            treePos = Vector3.zero;
            return false;
        }                                            

        Vector3 GetRandomTerrainPosition()
        {
            float terrainWidth = terrain.terrainData.size.x;
            float terrainLength = terrain.terrainData.size.z;

            float x = Random.Range(padding, terrainWidth - padding);
            float z = Random.Range(padding, terrainLength - padding);
            float y = terrain.SampleHeight(new Vector3(x, 0, z)) + terrain.transform.position.y;
            return new Vector3(terrain.transform.position.x + x, y, terrain.transform.position.z + z);
        }
                      

        bool IsFarFromResources(Vector3 position, float minDistance)
        {
            foreach (Vector3 existingPos in resourcePositions)
            {
                if (VecStor3.Distance(position, existingPos) < minDistance)
                {
                    return false;
                }
            }
            return true;
        }
    }    
                                        


    //Worker2.cs
    using UnityEngine;
    using UnityEngine.UI;
    using UnityEngine.AI;
    using System.Collections.Generic;
    using System.Linq;

    public enum WorkerState
    {
        Idle,
        Collecting,
        Depositing,
        Building,
        Moving
    }

    public class Worker2 : MonoBehaviour, IDamageable
    {
        [Header("Worker Configuration")]
        public int carryCapacity = 50;
        private int currentCarry = 0;
        private ResourceType2 carriedResource;
        private NavMeshAgent agent;
        private Animator animator;

        public GameObject storageDepotPrefab;
        private Vector3 buildTarget;

        private StorageDepot2 storageDepot;

        private WorkerState currentState = WorkerState.Idle;

        public WorkerState CurrentState
        {
            get { return currentState; }
            private set { currentState = value; UpdateAnimation(); }


        }

        [Header("Collection Configuration")]
        public float collectionRadius = 5f;

        [Header("Movement Configuration")]
        public float reachedDestinationThreshold = 5f;

        private Vector3 depositPoint;

        private bool isAssignedToCollect = false;

        private List<GameObject> availableResourceDeposits = new List<GameObject>();

        private GameObject targetResourceDeposit = null;

        [Header("Health Settings")]
        public float maxHealth = 100f;
        private float currentHealth;

        [Header("Health Bar UI")]
        public Slider healthBarSlider;

        [Header("Health Bar Positioning")]
        public Vector3 healthBarOffset = new Vector3(0, 2.5f, 0);

        [Header("Transition Settings")]
        public float smoothSpeed = 5f;

        private Transform cameraTransform;

        [Header("Selection VFX")]
        [Tooltip("Drag your circle‐VFX prefab here.")]
        public GameObject selectionVFXPrefab;
        private GameObject selectionVFXInstance;

        [Header("Deposit Configuration")]
        [SerializeField] private float depositRange = 10f;
        private bool waitingForFreeDepot = false;
        

        void Awake()
        {
            currentHealth = maxHealth;
            cameraTransform = Camera.main.transform;

            if (healthBarSlider == null)
            {
                Debug.LogError("HealthBarSlider is not assigned in the Worker2 script on " + gameObject.name);
            }

            if (healthBarSlider != null)
            {
                Transform healthBarCanvas = transform.Find("HealthBarCanvas");
                if (healthBarCanvas != null)
                {
                    healthBarCanvas.localPosition = healthBarOffset;
                }
                else
                {
                    Debug.LogWarning("HealthBarCanvas not found in " + gameObject.name);
                }
            }

            animator = GetComponentInChildren<Animator>();
            if (animator == null)
            {
                Debug.LogError("No Animator component found in children of " + gameObject.name);
            }

            if (selectionVFXPrefab != null)
            {
                selectionVFXInstance = Instantiate(
                    selectionVFXPrefab,
                    transform.position + Vector3.down * 1f,
                    Quaternion.identity,
                    transform
                );
                selectionVFXInstance.SetActive(false);
            }
            else
            {
                Debug.LogError("Selection VFX prefab is not assigned in the Worker2 script on " + gameObject.name);
            }

        }
        

        void Start()
        {
            agent = GetComponent<NavMeshAgent>();
            agent.stoppingDistance = depositRange;
            if (agent == null)
            {
                Debug.LogError($"Worker2 [{gameObject.name}]: NavMeshAgent component is missing.");
            }

            StorageDepot2[] depots = FindObjectsOfType<StorageDepot2>();
            if (depots.Length > 0)
            {
                storageDepot = depots[0];
                float minDist = Vector3.Distance(transform.position, storageDepot.transform.position);
                foreach (StorageDepot2 depot in depots)
                {
                    float dist = Vector3.Distance(transform.position, depot.transform.position);
                    if (dist < minDist)
                    {
                        storageDepot = depot;
                        minDist = dist;
                    }
                }
                depositPoint = storageDepot.transform.position;
                Debug.Log($"Worker2 [{gameObject.name}]: Found StorageDepot2 at {depositPoint}.");
            }
            else
            {
                Debug.LogError($"Worker2 [{gameObject.name}]: No StorageDepot2 found in the scene.");
            }

            RefreshAvailableResourceDeposits();
        }
                                        

        void Update()
        {
            if (healthBarSlider != null)
            {
                healthBarSlider.value = Mathf.Lerp(healthBarSlider.value, currentHealth / maxHealth, Time.deltaTime * smoothSpeed);
            }

            if (healthBarSlider != null && cameraTransform != null)
            {
                healthBarSlider.transform.parent.LookAt(cameraTransform);
            }

            HandleState();
            UpdateAnimation();
        }
                                        

        public void ShowSelectionCircle()
        {
            if (selectionVFXInstance != null)
                selectionVFXInstance.SetActive(true);
            GetComponent<WorkerAudioController>()?.PlayHi();

        }
        

        public void HideSelectionCircle()
        {
            if (selectionVFXInstance != null)
                selectionVFXInstance.SetActive(false);
        }
                                        

        private void HandleState()
        {
            switch (currentState)
            {
                case WorkerState.Idle:
                    UpdateAnimation();
                    if (targetResourceDeposit == null)
                    {
                        FindNearestResourceDeposit();
                    }
                    break;

                case WorkerState.Collecting:
                    UpdateAnimation();
                    if (targetResourceDeposit != null)
                    {
                        CollectResource();
                    }
                    else
                    {
                        Debug.Log($"Worker2 [{gameObject.name}]: No target resource deposit found. Switching to Idle.");
                        CurrentState = WorkerState.Idle;
                        UpdateAnimation();

                    }
                    break;

                case WorkerState.Depositing:
                    UpdateAnimation();
                    float dist = Vector3.Distance(transform.position, depositPoint);

                    if (dist <= depositRange)
                    {
                        if (waitingForFreeDepot)
                        {
                            if (CanStorageDepotAccept(storageDepot, carriedResource, currentCarry))
                            {
                                waitingForFreeDepot = false;
                                DepositResources();
                            }
                        }
                        else
                        {
                            DepositResources();
                        }
                    }
                    break;


                case WorkerState.Building:
                    UpdateAnimation();
                    if (!agent.pathPending && agent.remainingDistance <= 10f)
                    {
                        var audioController = GetComponent<WorkerAudioController>();
                        audioController?.PlayHammer();

                        Instantiate(storageDepotPrefab, buildTarget + Vector3.up * 2.5f, Quaternion.identity);
                        Debug.Log($"Worker2 [{gameObject.name}]: Built new StorageDepot at {buildTarget}.");
                        CurrentState = WorkerState.Idle;
                        UpdateAnimation();

                    }
                    break;

                case WorkerState.Moving:
                    UpdateAnimation();
                    if (!agent.pathPending && agent.remainingDistance <= reachedDestinationThreshold)
                    {
                        Debug.Log($"Worker2 [{gameObject.name}]: Reached destination.");
                        if (isAssignedToCollect && targetResourceDeposit != null)
                        {
                            CurrentState = WorkerState.Collecting;
                            UpdateAnimation();

                            Debug.Log($"Worker2 [{gameObject.name}]: Starting to collect resources.");
                        }
                        else
                        {
                            CurrentState = WorkerState.Idle;
                            UpdateAnimation();

                            Debug.Log($"Worker2 [{gameObject.name}]: Now Idle after moving.");
                        }
                    }
                    break;
            }
        }
                                        

        private void RefreshAvailableResourceDeposits()
        {
            availableResourceDeposits.Clear();
            GameObject[] resourceObjects = GameObject.FindGameObjectsWithTag("ResourceDeposit");
            foreach (GameObject obj in resourceObjects)
            {
                ResourceDeposit2 resourceData = obj.GetComponent<ResourceDeposit2>();
                if (resourceData != null && resourceData.totalResource > 0)
                {
                    availableResourceDeposits.Add(obj);
                }
            }

            Debug.Log($"Worker2 [{gameObject.name}]: Found {availableResourceDeposits.Count} available ResourceDeposits.");
        }
                                        

        private void FindNearestResourceDeposit()
        {
            RefreshAvailableResourceDeposits();

            GameObject nearest = null;
            float minDist = Mathf.Infinity;
            Vector3 currentPos = transform.position;

            foreach (GameObject resource in availableResourceDeposits)
            {
                ResourceDeposit2 resourceData = resource.GetComponent<ResourceDeposit2>();
                if (resourceData == null)
                    continue;

                float dist = Vector3.Distance(currentPos, resource.transform.position);
                if (dist < minDist)
                {
                    minDist = dist;
                    nearest = resource;
                }
            }

            if (nearest != null)
            {
                AssignCollect(nearest.GetComponent<ResourceDeposit2>());
            }
            else
            {
                Debug.Log($"Worker2 [{gameObject.name}]: No available resource deposits found.");
                CurrentState = WorkerState.Idle;
                UpdateAnimation();

            }
        }
                                        

        public void AssignCollect(ResourceDeposit2 deposit)
        {
            if (deposit == null)
            {
                Debug.LogError($"Worker2 [{gameObject.name}]: AssignCollect received a null deposit.");
                return;
            }

            if (currentCarry > 0 && carriedResource != deposit.resourceType)
            {
                Debug.Log($"Worker2 [{gameObject.name}]: Carrying {carriedResource}, needs to deposit before collecting {deposit.resourceType}.");
                AssignDeposit();
                targetResourceDeposit = deposit.gameObject; // Queue the new resource deposit after depositing
                isAssignedToCollect = true;
                return;
            }

            if (currentCarry >= carryCapacity)
            {
                Debug.Log($"Worker2 [{gameObject.name}]: Carry capacity reached, need to deposit before collecting more.");
                AssignDeposit();
                targetResourceDeposit = deposit.gameObject; // Queue the new resource deposit after depositing
                isAssignedToCollect = true;
                return;
            }

            targetResourceDeposit = deposit.gameObject;
            carriedResource = deposit.resourceType;

            agent.SetDestination(targetResourceDeposit.transform.position);
            currentState = WorkerState.Moving;
            UpdateAnimation();

            isAssignedToCollect = true;
            Debug.Log($"Worker2 [{gameObject.name}]: Assigned to collect {carriedResource} from {targetResourceDeposit.name}.");
        }
        

        public void AssignMove(Vector3 destination)
        {
            if (agent == null)
            {
                Debug.LogError($"Worker2 [{gameObject.name}]: NavMeshAgent is missing.");
                return;
            }

            if (currentState == WorkerState.Collecting || currentState == WorkerState.Depositing)
            {
                Debug.Log($"Worker2 [{gameObject.name}]: Interrupting current task to move to {destination}.");
            }

            isAssignedToCollect = false;
            targetResourceDeposit = null;

            agent.SetDestination(destination);
            currentState = WorkerState.Moving;
            UpdateAnimation();

            Debug.Log($"Worker2 [{gameObject.name}]: Assigned to move to {destination}.");
        }
                                        

        public void AssignBuild(Vector3 point)
        {
            buildTarget = point;
            agent.SetDestination(buildTarget);
            CurrentState = WorkerState.Building;
            UpdateAnimation();

            Debug.Log($"{gameObject.name}: Assigned to build depot at {buildTarget}.");
        }
                                        

        private void AssignDeposit()
        {
            StorageDepot2[] depots = FindObjectsOfType();
            if (depots.Length == 0)
            {
                Debug.LogError($"Worker2 [{name}]: No StorageDepot2 in scene to deposit.");
                CurrentState = WorkerState.Idle;
                return;
            }

            var sorted = depots.OrderBy(d => Vector3.Distance(transform.position, d.transform.position)).ToArray();

            StorageDepot2 chosen = sorted.FirstOrDefault(d =>
                CanStorageDepotAccept(d, carriedResource, currentCarry)
            );

            if (chosen != null)
            {
                waitingForFreeDepot = false;
            }
            else
            {
                chosen = sorted[0];
                waitingForFreeDepot = true;
                Debug.LogWarning($"Worker2 [{name}]: All depots full. Waiting at closest ({chosen.name}).");
            }

            storageDepot = chosen;
            depositPoint = storageDepot.transform.position;
            agent.SetDestination(depositPoint);
            CurrentState = WorkerState.Depositing;
            Debug.Log($"Worker2 [{name}]: Assigned to deposit at {storageDepot.name}. Waiting? {waitingForFreeDepot}");
        }
                                        

        private void DepositResources()
        {
            if (storageDepot == null)
            {
                Debug.LogError($"Worker2 [{gameObject.name}]: Cannot deposit resources because StorageDepot2 is missing.");
                return;
            }

            if (currentCarry > 0)
            {
                if (CanStorageDepotAccept(storageDepot, carriedResource, currentCarry))
                {
                    storageDepot.Deposit(carriedResource, currentCarry);
                    Debug.Log($"Worker2 [{gameObject.name}]: Deposited {currentCarry} {carriedResource} at StorageDepot2.");
                    currentCarry = 0;
                    carriedResource = default(ResourceType2);

                    if (isAssignedToCollect && targetResourceDeposit != null)
                    {
                        AssignCollect(targetResourceDeposit.GetComponent<ResourceDeposit2>());
                    }
                    else
                    {
                        CurrentState = WorkerState.Idle;
                        UpdateAnimation();

                        Debug.Log($"Worker2 [{gameObject.name}]: Deposit completed. Now Idle.");
                    }
                }
                else
                {
                    Debug.LogWarning($"Worker2 [{gameObject.name}]: StorageDepot2 is full. Cannot deposit {carriedResource} now. Becoming Idle.");
                    CurrentState = WorkerState.Idle;
                    UpdateAnimation();

                }
            }
            else
            {
                Debug.Log($"Worker2 [{gameObject.name}]: No resources to deposit.");
                CurrentState = WorkerState.Idle;
                UpdateAnimation();

            }
        }
        

        private void CollectResource()
        {
            if (targetResourceDeposit == null)
            {
                Debug.LogError($"Worker2 [{gameObject.name}]: No target resource deposit to collect from.");
                CurrentState = WorkerState.Idle;
                UpdateAnimation();

                return;
            }

            ResourceDeposit2 resourceData = targetResourceDeposit.GetComponent<ResourceDeposit2>();
            if (resourceData == null)
            {
                Debug.LogError($"Worker2 [{gameObject.name}]: ResourceDeposit2 component missing on {targetResourceDeposit.name}.");
                CurrentState = WorkerState.Idle;
                UpdateAnimation();

                return;
            }

            float distance = Vector3.Distance(transform.position, targetResourceDeposit.transform.position);
            if (distance > collectionRadius)
            {
                Debug.Log($"Worker2 [{gameObject.name}]: Moved out of collection radius for {targetResourceDeposit.name}. Assigning deposit.");

                return;
            }

            int amountToCollect = carryCapacity - currentCarry;
            int collectedAmount = resourceData.Collect(amountToCollect);

            if (collectedAmount > 0)
            {
                currentCarry += collectedAmount;
                carriedResource = resourceData.resourceType;
                Debug.Log($"Worker2 [{gameObject.name}]: Collected {collectedAmount} {carriedResource} from {targetResourceDeposit.name}.");

                var audioController = GetComponent<WorkerAudioController>();
                switch (carriedResource)
                {
                    case ResourceType2.Wood:
                        audioController?.PlayWoodChop();
                        break;
                    case ResourceType2.Stone:
                        audioController?.PlayStonePick();
                        break;
                    case ResourceType2.Mana:
                        audioController?.PlayManaRing();
                        break;
                    case ResourceType2.Food:
                        audioController?.PlayCropRustle();
                        break;
                }

                AssignDeposit();
            }
            else
            {
                if (resourceData.totalResource <= 0)
                {
                    Debug.Log($"Worker2 [{gameObject.name}]: {carriedResource} node depleted. Searching next node.");
                    GameObject next = GetNearestResourceDepositOfType(carriedResource);
                    if (next != null)
                    {
                        AssignCollect(next.GetComponent<ResourceDeposit2>());
                        return;
                    }
                }

                Debug.Log($"Worker2 [{gameObject.name}]: No resources to collect from {resourceData.resourceType}. Depositing/Idling.");
                if (currentCarry > 0)
                    AssignDeposit();
                else
                    CurrentState = WorkerState.Idle;
                UpdateAnimation();

            }
        }
                                        

        private bool CanStorageDepotAccept(StorageDepot2 depot, ResourceType2 type, int amount)
        {
            int current = depot.GetTotalResource(type);
            switch (type)
            {
                case ResourceType2.Stone: return current + amount <= StorageDepot2.MaxStoneStorage;
                case ResourceType2.Mana: return current + amount <= StorageDepot2.MaxManaStorage;
                case ResourceType2.Food: return current + amount <= StorageDepot2.MaxFoodStorage;
                case ResourceType2.Wood: return current + amount <= StorageDepot2.MaxWoodStorage;
                default:
                    Debug.LogWarning($"Worker2 [{name}]: Unknown ResourceType2 {type}.");
                    return false;
            }
        }
        

        public void TakeDamage(float amount, GameObject attacker)
        {
            currentHealth -= amount;
            currentHealth = Mathf.Clamp(currentHealth, 0f, maxHealth);
            Debug.Log($"{gameObject.name} took {amount} damage from {attacker.name}. Current Health: {currentHealth}");

            if (healthBarSlider != null)
            {

            }
            else
            {
                Debug.LogWarning("Worker2: HealthBarSlider is null.");
            }

            if (currentHealth <= 0)
            {
                Die(attacker);
            }
        }
                                        

        private void Die(GameObject attacker)
        {

        private GameObject GetNearestResourceDepositOfType(ResourceType2 type)
        {
            RefreshAvailableResourceDeposits();
            GameObject nearest = null;
            float minDist = Mathf.Infinity;
            Vector3 myPos = transform.position;

            foreach (GameObject obj in availableResourceDeposits)
            {
                var data = obj.GetComponent<ResourceDeposit2>();
                if (data == null || data.resourceType != type) continue;
                if (obj == targetResourceDeposit) continue;

                float d = Vector3.Distance(myPos, obj.transform.position);
                if (d < minDist)
                {
                    minDist = d;
                    nearest = obj;
                }
            }
            return nearest;
        }
